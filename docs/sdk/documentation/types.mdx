# Types

## Product types

```js
type Product = {
  id: number,
  title: string,
  description: string,
  sku?: string,
  qty: number,
  images?: { url: string }[],
  publicPrice?: string,
  optionsEnabled?: boolean,
  options?: ProductOption[],
  variants?: ProductVariant[],
  price: {
    amount: string,
    currencyCode: string,
  },
};

type ProductOption = {
  id: number,
  deletedAt: string,
  name: string,
  order: 1,
  values: string,
};

type ProductVariant = {
  barcode: string,
  deletedAt: string,
  id: number,
  originId: string,
  originalPrice: { amount: string, currencyCode: string },
  price: string,
  quantity: number,
  sku: string,
  title: string,
};
```

```js
interface Cart {
  cart_id: string;
  customer_session_id: string;
  created_at: Date;
  updated_at: Date;
  line_items: {
    product_id: number,
    variant_id: number | null | undefined,
    quantity: number,
    price_data: {
      unit_price: number,
      tax: number,
      currency: string,
    },
  }[];
  total_amount: number;
  currency: string;
}
```
